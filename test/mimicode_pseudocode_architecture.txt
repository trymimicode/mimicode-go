MIMICODE GO ARCHITECTURE PSEUDOCODE
=====================================

OVERVIEW:
Mimicode is an AI coding assistant that uses Claude (Anthropic) to help with programming tasks.
It provides both REPL and TUI interfaces, manages conversation sessions, and includes various
coding tools like file editing, bash execution, and web search.

MAIN COMPONENTS:

function run_repl():
    session = create_or_resume_session()
    while true:
        prompt = read_user_input()
        if prompt == "exit":
            break
        
        response = agent_turn(prompt)
        display_response(response)
        
        maybe_compact_session()  # Auto-compress old messages
        save_session()

function run_oneshot():
    session = create_session()
    response = agent_turn(prompt)
    display_response(response)
    reflect_on_session()  # Generate memory entries

    bash: execute shell commands

    write: create/overwrite files

    web_fetch: fetch URL content
    stackoverflow_search: search Stack Overflow

function agent_turn(user_message, config):
    messages = session.get_messages()
    messages.append(user_message)
    
    # Add system context
    if repo_map_exists():
        messages[0].content.append(repo_map)
    if memory_exists():
        messages[0].content.append(memory_content)
    
    step_count = 0
    while step_count < MAX_STEPS:
        # Call Claude API
        response = call_claude(messages, tools=TOOLS)
        
        # Process response
        if response.has_tool_calls():
            for tool_call in response.tool_calls:
                result = execute_tool(tool_call)
                messages.append(tool_result_message(result))
            step_count++
        else:
            # Final response reached
            break
    
    return response

function execute_tool(tool_call):
    switch tool_call.name:
        case "bash":
            return bash_execute(tool_call.input)
        case "read":
            return read_file(tool_call.input)
        case "edit":
            return edit_file(tool_call.input)
        # ... other tools

function bash_execute(cmd, timeout):
    # Vet dangerous commands
    for rule in BLOCKED_PATTERNS:
        if matches(cmd, rule):
            return error(rule.hint)
    
    # Execute with timeout
    process = start_process(cmd)
    result = wait_with_timeout(process, timeout)
    
    # Strip ANSI codes and truncate
    output = strip_ansi(result.output)
    if len(output) > MAX_OUTPUT_SIZE:
        output = tail_truncate(output)
        
    return ToolResult{output, is_error, truncated}

function read_file(path, offset, limit):
    if is_binary(path):
        return error("binary file")
        
    content = read_file_lines(path)
    
    # Apply offset/limit
    if offset > 0:
        content = content[offset:]
    if limit > 0:
        content = content[:limit]
        
    return format_with_line_numbers(content)

function edit_file(path, edits):
    # Lock file for concurrent access
    lock = get_file_lock(path)
    lock.acquire()
    defer lock.release()
    
    content = read_file(path)
    
    # Apply all edits atomically
    for edit in edits:
        matches = find_exact_matches(content, edit.old_text)
        if len(matches) != 1:
            return error("need exactly one match")
        content = replace(content, edit.old_text, edit.new_text)
    
    write_file(path, content)
    return generate_diff(old_content, content)

function maybe_compact(session):
    config = get_compact_config()
    messages = session.get_messages()
    
    # Check turn count and token thresholds
    if should_compact_based_on_thresholds():
        compact_messages(messages)

function compact_messages(messages):
    # Keep recent messages intact
    split_point = find_split(messages, keep_recent=3)
    
    to_compress = messages[:split_point]
    to_keep = messages[split_point:]
    
    # Summarize with Haiku (fast/cheap model)
    summary = call_claude_haiku(
        prompt=COMPACTION_PROMPT,
        transcript=flatten_messages(to_compress)
    )
    
    # Create marker message
    marker = create_compaction_marker(summary, to_compress)
    
    # Replace old messages with marker
    return [marker] + to_keep

9. REFLECTION (internal/reflect/)
----------------------------------------
function reflect_on_session(session):
    messages = session.get_messages()
    
    # Extract work done
    files_changed = extract_file_changes(messages)
    decisions_made = extract_decisions(messages)
    
    if files_changed.empty() and decisions_made.empty():
        return  # Nothing meaningful to remember
    
    # Generate reflection prompt
    prompt = create_reflection_prompt(messages)
    
    # Call Claude to generate memory entries
    reflection = call_claude(prompt)
    
    # Write to MEMORY.md
    for entry in reflection.memory_entries:
        memory_write(entry.component, entry.summary, entry.details)
